[LazyEvaluation]

var
  ElseNode: INode
begin
  if aNode.Child[2].AsBoolean then //if aNode.Child['expr'].AsBoolean
    Result := aNode.Child[4].Call
  else
  begin
    ElseNode = aNode.Child[5];

  end;
end;
